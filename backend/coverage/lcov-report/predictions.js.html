const express = require("express");
const Joi = require("joi");
const { getTeamStats, getStandings } = require("../services/footballApi");
const { predict, getModelStats, getPredictionHistory, savePrediction } = require("../services/mlService");
 
const router = express.Router();
 
// ══════════════════════════════════════════════════════════════
// VALIDATION SCHEMA (Improvement #9)
// ══════════════════════════════════════════════════════════════
const predictSchema = Joi.object({
  homeTeamId: Joi.number().integer().positive().required(),
  awayTeamId: Joi.number().integer().positive().required(),
  competition: Joi.string().uppercase().valid("PL", "SA", "PD", "BL1").required(),
});
 
// Competition flags removed as they have 0 importance (Improvement #4)
// Model v2 doesn't use these features
 
/**
 * POST /api/predictions/predict
 * Body: { homeTeamId, awayTeamId, competition }
 *
 * IMPROVED: Now includes ALL features the model needs:
 * - Form stats (home/away)
 * - Rank and rank_diff
 * - Venue-specific goals
 * - Head-to-head stats
 */
router.post("/predict", async (req, res) => {
  // Validate request (Improvement #9)
  const { error, value } = predictSchema.validate(req.body);
  if (error) {
    return res.status(400).json({
      error: error.details[0].message,
      validCompetitions: ["PL", "SA", "PD", "BL1"],
    });
  }
 
  const { homeTeamId, awayTeamId, competition } = value;
 
  try {
    // 1. Fetch team stats in parallel
    const [homeStats, awayStats, standings] = await Promise.all([
      getTeamStats(Number(homeTeamId)),
      getTeamStats(Number(awayTeamId)),
      getStandings(competition),
    ]);
 
    // 2. Calculate rank from standings (Improvement #6 - missing features)
    const homeRank = standings.table.find(t => t.teamId === Number(homeTeamId))?.position || 10;
    const awayRank = standings.table.find(t => t.teamId === Number(awayTeamId))?.position || 10;
    const rankDiff = awayRank - homeRank; // Positive = home team ranked higher
 
    // 3. Calculate venue-specific goals (approximation from recent form)
    // In a full implementation, you'd track home/away splits separately
    const homeGoalsScoredHome = homeStats.form_gf * 1.1;  // Home teams score ~10% more at home
    const homeGoalsScoredAway = homeStats.form_gf * 0.9;
    const awayGoalsScoredHome = awayStats.form_gf * 1.1;
    const awayGoalsScoredAway = awayStats.form_gf * 0.9;
 
    // 4. Head-to-head (simplified - in production, fetch actual H2H data)
    // Default to balanced H2H if no data
    const h2hHomeWins = 1.0;
    const h2hAwayWins = 1.0;
 
    // 5. Build COMPLETE feature set matching model v2
    const matchFeatures = {
      // Form features (existing)
      home_form_pts:  homeStats.form_pts,
      home_form_gf:   homeStats.form_gf,
      home_form_ga:   homeStats.form_ga,
      away_form_pts:  awayStats.form_pts,
      away_form_gf:   awayStats.form_gf,
      away_form_ga:   awayStats.form_ga,
      form_pts_diff:  parseFloat((homeStats.form_pts - awayStats.form_pts).toFixed(3)),
      form_gf_diff:   parseFloat((homeStats.form_gf - awayStats.form_gf).toFixed(3)),
      form_ga_diff:   parseFloat((homeStats.form_ga - awayStats.form_ga).toFixed(3)),
      
      // Rank features (NEW - Improvement #6)
      home_rank:      homeRank,
      away_rank:      awayRank,
      rank_diff:      rankDiff,
      
      // Venue-specific goals (NEW - Improvement #6)
      home_goals_scored_home: parseFloat(homeGoalsScoredHome.toFixed(3)),
      home_goals_scored_away: parseFloat(homeGoalsScoredAway.toFixed(3)),
      away_goals_scored_home: parseFloat(awayGoalsScoredHome.toFixed(3)),
      away_goals_scored_away: parseFloat(awayGoalsScoredAway.toFixed(3)),
      
      // Head-to-head (NEW - Improvement #6)
      h2h_home_wins:  h2hHomeWins,
      h2h_away_wins:  h2hAwayWins,
    };
 
    // 6. Get prediction from ML service
    const prediction = await predict(matchFeatures);
 
    // 7. Build response with enhanced data
    const response = {
      prediction: {
        ...prediction,
        // Add confidence percentage for visualization (Improvement #11)
        confidencePercent: Math.round(
          Math.max(prediction.probability_home_win, prediction.probability_away_win) * 100
        ),
      },
      features: matchFeatures,
      homeTeam: {
        id: homeTeamId,
        name: homeStats.teamName,
        crest: homeStats.teamCrest,
        rank: homeRank,
        stats: homeStats,
      },
      awayTeam: {
        id: awayTeamId,
        name: awayStats.teamName,
        crest: awayStats.teamCrest,
        rank: awayRank,
        stats: awayStats,
      },
      competition,
      timestamp: new Date().toISOString(),
    };
 
    // 8. Save to prediction history (Improvement #10)
    try {
      await savePrediction(response);
    } catch (e) {
      console.warn("[predictions] Failed to save to history:", e.message);
    }
 
    res.json(response);
  } catch (err) {
    console.error("[predictions] Error:", err.message);
    const status = err.message.includes("ML service") ? 502 : 500;
    res.status(status).json({ error: err.message });
  }
});

  try {
    const limit = parseInt(req.query.limit) || 20;
    const history = await getPredictionHistory(limit);
    res.json({ predictions: history, count: history.length });
  } catch (err) {
    res.status(500).json({ error: err.message });
